\begin{equation}
    \forall A = (a_1,\ldots,a_N) \in \mathbb{Z}^N:
\quad
N \equiv 1 \mod(2)
\quad\Rightarrow\quad
\forall i \in [1,N] \subset \mathbb{Z}:
\quad
\left(
a_i = M(A) \Leftrightarrow i = \frac{N+1}{2}
\right)
\end{equation}
